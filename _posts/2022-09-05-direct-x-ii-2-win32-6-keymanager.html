### 1. KeyManager 클래스의 필요성

Win32 내장함수인 GetAsyncKeyState로 얻을 수 있는 키의 상태는 제한적이다. 그냥 함수를 호출하는 그 순간에 키가 눌려있는지 여부만 알려주기 때문에,
키가 최초로 눌린 건지, 이전부터 눌려있던 건지 알 수 없고 키가 떼어지는 순간도 체크할 수 없다.
따라서 1. 키의 눌림, 2. 키의 홀드(눌려있는 상태), 3. 키의 떼어짐, 이렇게 세 가지 상태를 매 프레임마다 갱신하고 저장하는 클래스를 만들어줄 것이다.

저번엔 GetAsyncKeyState() 함수로 현재 키가 눌렸는지 체크하고, 이전에 눌렸는지는 전역변수 g_prev_pressed 따위로 정의했다.
이것을 게임에서 사용할 모든 키에 대해 저장하고 KeyManager 클래스에서 관리하도록 했다.

### 2. KeyManager 클래스 구조

벡터 자료구조를 사용했다.

> 단순히 키-키의 상태를 매핑할 생각으로 처음엔 맵 자료구조를 쓰는 게 낫지 않은가 생각했는데,
키의 개수는 많아야 100개를 넘지 않고 삽입 삭제는 이루어지지 않으며 잦은 탐색이 이루어진다는 걸 고려하니 벡터를 사용한 이유를 알 것 같았다.
맵 자료구조는 임의의 위치에 삽입, 삭제, 탐색이 모두 O(log n) 시간에 이루어진다.
벡터 자료구조는 맨 뒤에 삽입과 삭제는 O(1), 임의의 위치에 삽입과 삭제는 O(n), 탐색은 O(1) 시간에 이루어진다.
초기화할 때는 맨 뒤에 삽입, 이후 프로그램이 돌아가는 동안은 탐색만 이루어지므로 벡터를 사용할 경우 O(1) 시간을 보장한다.

```c++
//KeyManager.h
#pragma once
enum class KEY_STATE {
	UP, HOLD, DOWN, NONE
};

enum class KEY {
	UP,
	DOWN,
	LEFT,
	RIGHT,
	SHIFT,
	CTRL,
	ALT,
	Q,
	W,
	E,
	R,
	T,
	Y,
	U,
	I,
	O,
	P,
	A,
	S,
	D,
	F,
	G,
	H,
	J,
	K,
	L,
	Z,
	X,
	C,
	V,
	B,
	N,
	M,
	LAST, //마지막 체크용
};


struct CurrentKeyState {
	KEY_STATE state;
	bool prev_pressed;
};
class KeyManager
{
	SINGLETON(KeyManager);
private:
	std::vector<CurrentKeyState> current_key_states;
public:
	bool Init();
	bool Update();
	KEY_STATE get_key_states(KEY key);
	int ToWindowsKey(KEY key);
};


```

키의 상태를 저장할 enum 클래스를 하나 만든다.

> enum과 enum class의 차이: enum은 서로 다른 enum 안에서 같은 이름의 식별자를 사용하면 컴파일 에러가 나서, namespace를 따로 만들어줘야 하는 단점이 있다.
하지만 enum class는 서로 다른 enum class 안에서 같은 이름의 식별자를 사용하는 것이 가능하다.

키가 최초로 눌릴 경우 DOWN, 이전 프레임부터 눌린 상태일 경우 HOLD, 키가 떨어졌을 경우 UP을, 이전 프레임에도 현재 프레임에도 눌리지 않았을 경우 NONE으로 설정한다.

current_key_states라는 벡터 자료구조에 KEY enum class의 int 값을 인덱스로 CurrentKeyState 구조체를 저장한다.
앞으로 current_key_states[KEY]에 해당 키 상태가 저장된다.

또한 다른 클래스에서 키의 상태만을 얻기 위해 get_key_states 라는 함수를 따로 정의해준다.

```c++
//KeyManager.cpp
#include "pch.h"
#include "KeyManager.h"

int g_windows_keys[static_cast<int>(KEY::LAST)] = {
	VK_UP,//UP,
	VK_DOWN,//DOWN,
	VK_LEFT,//LEFT,
	VK_RIGHT,//RIGHT,
	VK_SHIFT,//SHIFT,
	VK_CONTROL,//CTRL,
	VK_MENU,//ALT,
	'Q',//Q,
	'W',//W,
	'E',//E,
	'R',//R,
	'T',//T,
	'Y',//Y,
	'U',//U,
	'I',//I,
	'O',//O,
	'P',//P,
	'A',//A,
	'S',//S,
	'D',//D,
	'F',//F,
	'G',//G,
	'H',//H,
	'J',//J,
	'K',//K,
	'L',//L,
	'Z',//Z,
	'X',//X,
	'C',//C,
	'V',//V,
	'B',//B,
	'N',//N,
	'M',//M,
};
KeyManager::KeyManager() {}
KeyManager::~KeyManager() {}


bool KeyManager::Init() {
	
	//current_key_states 초기화
	for (int i=0; i < static_cast<int>(KEY::LAST); i++) {
		current_key_states.push_back(CurrentKeyState{ KEY_STATE::NONE, FALSE });
	}
	return TRUE;
}
bool KeyManager::Update()
{
	//프로그램이 포커싱됨
	if (GetFocus() != nullptr) {
		for (int i = 0; i < static_cast<int>(KEY::LAST); i++) {
			//키 눌린 상태
			if (GetAsyncKeyState(g_windows_keys[i]) & 0x8000) {
				//이전 프레임에서 눌리지 않음 -> 최초 입력(DOWN)
				if (!current_key_states[i].prev_pressed) {
					current_key_states[i].state = KEY_STATE::DOWN;
				}
				//이전 프레임에서도 눌림 -> HOLD
				else {
					current_key_states[i].state = KEY_STATE::HOLD;
				}
				current_key_states[i].prev_pressed = TRUE;
			}
			//키 눌리지 않은 상태
			else {
				//이전 프레임에서 눌림 -> 키 떨어짐(UP)
				if (current_key_states[i].prev_pressed) {
					current_key_states[i].state = KEY_STATE::UP;
				}
				//이전 프레임에서 눌리지 않음 -> NONE
				else {
					current_key_states[i].state = KEY_STATE::NONE;
				}
				current_key_states[i].prev_pressed = FALSE;
			}
		}
	}
	//프로그램이 포커싱되어 있지 않음 -> 자연스럽게 키가 떨어진 것처럼 바꿈
	else {
		for (int i = 0; i < static_cast<int>(KEY::LAST); i++) {
			//이전 프레임에서 눌렸었음 -> 떨어짐(UP) 상태로 변경
			if (current_key_states[i].prev_pressed) {
				current_key_states[i].state = KEY_STATE::UP;
			}
			//이전 프레임에서 눌리지 않았음 -> NONE 상태로 변경
			else {
				current_key_states[i].state = KEY_STATE::NONE;
			}
			current_key_states[i].prev_pressed = FALSE;
			
		}
	}
	
	return TRUE;
}

KEY_STATE KeyManager::get_key_states(KEY key) { return current_key_states[static_cast<int>(key)].state; }
int KeyManager::ToWindowsKey(KEY key) { return g_windows_keys[static_cast<int>(key)]; }
```
Init 함수에선 current_key_states 벡터를 초기화해주고,
Update 함수에선 current_key_states 벡터를 순회하며 값을 갱신해준다.

### 3. 프로그램에 적용

먼저, RECT 자료구조는 Long 타입이기 때문에 소숫점을 가진 Rect2 구조체를 Vector2와 함께 새로 만들고,

```c++
struct Vector2 {
	float x;
	float y;
};

struct Rect2 {
	float left;
	float top;
	float right;
	float bottom;
};
```

GObject.h에도 몇가지 정의를 추가했다. 오브젝트가 rect와 pos로 이루어지도록 했다.
set_pos()로 pos를 변경하면 rect도 동기화된다.

```c++
//GObject.h
class GObject{
	//...생략
private:
	Rect2 rect_;
	Vector2 pos_;
	//...생략
public:
	inline Rect2 get_rect() { return rect_; };
	inline float get_width() { return abs(rect_.right - rect_.left); };
	inline float get_height() { return abs(rect_.bottom - rect_.top); };
	inline Vector2 get_pos() { return pos_; }
	inline void set_pos(Vector2 v) { //rect도 동기화
		float x_axis_dist = v.x - get_pos().x;
		float y_axis_dist = v.y - get_pos().y;
		pos_ = v;
		rect_.left += x_axis_dist;
		rect_.right += x_axis_dist;
		rect_.top += y_axis_dist;
		rect_.bottom += y_axis_dist;
	}
```

단순하게 네모를 이동시키는 코드를 작성해봤다.

```c++
//Core.cpp
GObject g_gobj( Rect2{ 500, 500, 510, 510 } );
bool Core::Update()
{
	Time::GetInstance()->Update();
	KeyManager::GetInstance()->Update();
	Vector2 v = g_gobj.get_pos();
	if (KeyManager::GetInstance()->get_key_states(KEY::D) == KEY_STATE::HOLD
		|| KeyManager::GetInstance()->get_key_states(KEY::RIGHT) == KEY_STATE::HOLD) {
		v.x += .2f* Time::GetInstance()->dt_f();

	}
	if (KeyManager::GetInstance()->get_key_states(KEY::A) == KEY_STATE::HOLD
		|| KeyManager::GetInstance()->get_key_states(KEY::LEFT) == KEY_STATE::HOLD) {
		v.x -= .2f * Time::GetInstance()->dt_f();

	}
	if (KeyManager::GetInstance()->get_key_states(KEY::W) == KEY_STATE::HOLD
		|| KeyManager::GetInstance()->get_key_states(KEY::UP) == KEY_STATE::HOLD) {
		v.y -= .2f * Time::GetInstance()->dt_f();

	}
	if (KeyManager::GetInstance()->get_key_states(KEY::S) == KEY_STATE::HOLD
		|| KeyManager::GetInstance()->get_key_states(KEY::DOWN) == KEY_STATE::HOLD) {
		v.y += .2f * Time::GetInstance()->dt_f();

	}
	g_gobj.set_pos(v);
	return TRUE;
}

bool Core::Render()
{
	Clear();
	Rect2 rect = g_gobj.get_rect();
	Rectangle(h_mem_dc_, rect.left, rect.top, rect.right, rect.bottom);
	BitBlt(h_dc_, 0, 0, pt_resolution_.x, pt_resolution_.y, h_mem_dc_, 0, 0, SRCCOPY);
	
	return TRUE;
}
```
