1. 적 스폰하기


매 웨이브마다 적들을 소환하고,
적들을 모두 잡으면 다음 웨이브로 넘어가게 할 것이다.



웨이브 정보에는 적의 종류와 수, 그리고 스탯이 들어간다.
웨이브가 올라갈수록 더 많고 강한 적이 나오게 할 것이다.

스포너는 <전체 웨이브 정보가 담긴 배열>과 <스폰해야 할 영역 정보>를 갖고 있다.
현재 웨이브에서 소환해야 하는 적 수만큼 랜덤 위치를 뽑아서 적을 소환한다. 이때 한 번에 모두 소환하지 않고 시간 간격을 두고 소환하게 한다.
그리고 플레이어가 모든 적을 처치하면 다음 웨이브로 넘긴다.


이런 느낌

(아직 게임 클리어는 구현하지 않음)


일단 스포너가 받아야 하는 입력을 정리해보면

1. 전체 웨이브 정보: 적의 종류, 적의 수, 적의 스탯
2. 한 웨이브 안에서 적을 소환할 시간 간격
3. 소환한 적이 모두 죽었는가(다음 웨이브로 넘기기 위해)
4. 소환가능한 영역

이렇게 되는데,


1. 전체 웨이브 정보

먼저 웨이브 정보는 Serializable 구조체로 정의해서 직접 값을 넣어줄 것이다.

이런 식으로


2. 한 웨이브 안에서 적을 소환할 시간 간격

한 웨이브에서 적을 소환할 시간 간격은 크게 중요하지 않으므로 멤버변수로 내부에서 설정해준다.



3. 소환한 적이 모두 죽었는지

먼저, 소환한 적이 죽었는지 알기 위해서는 적이 죽을 때마다 스포너에게 알려주거나, 스포너가 자기가 소환한 적들을 전부 배열로 갖고 있다가 살아있는지 매 프레임 체크하거나 여러 방법이 있겠지만, 적이 스포너 정보를 갖고 있는 건 좀 이상하게 느껴지고 스포너가 모든 적들을 매번 순회하며 체크하는 건 낭비가 심할 것 같다.

그래서 이벤트를 사용할 것이다.

적이 죽을 때마다 이벤트를 발행하고, 이 이벤트를 스포너가 구독하도록 한다.
"죽었다"라는 이벤트는 특정 적만이 아니라 LivingEntity라면 모두 갖고있도록 하는 게 자연스럽기 때문에 LivingEntity의 public 멤버변수로 선언한다.
파라미터로 죽은 객체를 받기로 한다.

4. 소환가능한 영역

소환 가능한 영역은 타원형으로 정의하는 게 가장 편리할 것 같다.

타원의 정의는 긴 반지름과 짧은 반지름, 그리고 원점 좌표로 이루어진다. 4개의 값만 정의하면 된다는 소리다.

하지만 숫자로 타원 구역을 정의해놓으면 한 눈에 알기 어렵다. 소환지점을 바꿔야 할 때마다 오브젝트를 하나 생성해서 이동시켜가며 현재 위치 좌표를 알아내고 반지름 크기만큼 키워서 가늠하고...너무 귀찮을 것 같다.

그래서 그냥 3D 오브젝트에서 제공하는 Cylinder를 생성하고 이 실린더를 스포너에게 넘겨주기로 했다.


이러면 눈에 보이게 영역을 정의할 수 있다.

실린더는 실린더 콜라이더를 가질 것 같았는데 캡슐 콜라이더를 갖더라.
캡슐 콜라이더여도 필요한 정보를 다 뽑아낼 수 있다.

스포너는 일단 캡슐 콜라이더를 public으로 받을 수 있게 한다.



일단 Random함수를 보면 Random.insindeUnitCircle이라는 프로퍼티가 있다.
반지름이 1인 원 안에서 랜덤 좌표를 반환하는 프로퍼티다.

(-1, 1) 범위의 Vector2 값이 나온다.
여기에 (콜라이더는 항상 y축에 평행하다고 가정할 것이므로) 콜라이더의 가로 반지름, 세로 반지름 값을 곱해주면 랜덤값을 콜라이더 전체 범위로 확장할 수 있다.

캡슐 콜라이더의 가로 세로 값은 반지름에 localScale을 곱하면 나온다.
원점도 가져올 수 있다.

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
 
public class EnemySpawner : MonoBehaviour
{
    public Wave[] Waves;        //***웨이브: 인덱스 0부터 저장
    public CapsuleCollider SpawnArea;  //스폰할 지역
 
    int _currentWave = 0;       //***현재 웨이브: 인덱스 1부터 셈.
    int _remainEnemyToSpawn;    //이번 웨이브에 스폰해야 할 적 수
    int _deadEnemyInWave;       //이번 웨이브에 죽은 적 수
    float _remainSpawnCooltime; // 스폰 주기
    const float cSpawnCooltime = 0.5f;       // 다음 스폰까지 남은 시간
 
 
    private void Start()
    {
        NextWave();
    }
 
    private void Update()
    {
        //아직 스폰해야 할 적이 남아있다면 일정 시간마다 스폰
        
        if (_remainEnemyToSpawn > 0) {
            _remainSpawnCooltime -= Time.deltaTime;
            if (_remainSpawnCooltime <= 0)
            {
                SpawnEnemy();
                _remainSpawnCooltime = cSpawnCooltime;
            }
        }
 
        //적을 모두 스폰했고 이번 웨이브에 죽은 적이 소환한 적 수와 같다면 => 다음 웨이브
        if (Waves.Length > _currentWave)
        {
            if (_remainEnemyToSpawn == 0 && _deadEnemyInWave == Waves[_currentWave - 1].EnemyAmount)
            {
                NextWave();
            }
        }
        
 
 
        
    }
    public void NextWave()
    {
        _currentWave++;
        Debug.Log("웨이브" + _currentWave + " 시작");
        _remainEnemyToSpawn = Waves[_currentWave - 1].EnemyAmount;
        _deadEnemyInWave = 0;
 
    }
    public void SpawnEnemy()
    {
 
        //랜덤 스폰지점(_spawnArea 타원 반경 안에서)
        Vector3 center = SpawnArea.transform.position;
        float xRadius = SpawnArea.radius * SpawnArea.transform.localScale.x;
        float zRadius = SpawnArea.radius * SpawnArea.transform.localScale.z;
 
        Vector2 randomBase = Random.insideUnitCircle;
        Vector3 randomPoint = new Vector3(center.x + xRadius * randomBase.x, center.y, center.z + zRadius * randomBase.y);
 
 
 
 
        // 해당 위치에 충돌체가 없을 때(대략 2미터 반경)
        LayerMask layerMask = LayerMask.NameToLayer("Item");
        if(Physics.OverlapSphere(randomPoint, 2, ~layerMask).Length == 0)
        {
            //랜덤 몬스터
            Wave currentWaveData = Waves[_currentWave - 1];
            int randomIndex = Random.Range(0, currentWaveData.EnemySpecies.Length);
            Enemy enemy = Instantiate(currentWaveData.EnemySpecies[randomIndex], new Vector3(randomPoint.x, 1, randomPoint.z), Quaternion.identity);
            enemy.Atk = currentWaveData.Atk;
            enemy.Def = currentWaveData.Def;
            enemy.Speed = currentWaveData.Speed;
            enemy.MaxHp = currentWaveData.maxHp;
            enemy.OnDeath += OnEnemyDeath;
 
            _remainEnemyToSpawn--;
        }
    }
    public void OnEnemyDeath(LivingEntity enemy)
    {
        _deadEnemyInWave++;
    }
 
    [System.Serializable]
    public struct Wave
    {
        public Enemy[] EnemySpecies;
        public int EnemyAmount;
        public int Atk;
        public int Def;
        public int Speed;
        public int maxHp;
    }
}

2. 적 공격 구현하기

지금은 한 종류의 적만 만들 것이다.
플레이어에게 다가와서 투사체를 날리는 적이다.

먼저, 플레이어에게 다가오는 건 유니티에서 제공하는 길찾기 AI인 NavMeshAgent를 사용했다.
맵으로 사용할 영역을 정한 뒤 맵에 있는 오브젝트들을 Walkable, Not Walkable, Jump 세 종류로 설정하고, 적 오브젝트에게 NavMeshAgent를 붙이고 목적지를 설정하면 최적의 경로를 찾아 자동으로 다가간다.
플레이어는 매 프레임 움직일 것이므로 매 프레임 갱신해야 한다.
하지만 경로를 찾는 건 비싼 연산이라 자주 실행하지 않는 게 좋다고 하기에 코루틴으로 정해진 시간마다 찾기로 했다.

[RequireComponent(typeof(NavMeshAgent))]
public class Enemy : LivingEntity
{
    public Projectile Projectile;
 
    const float cDistance = 5f;         // 최소 유지거리
    const float cAttackDistance = 10f;   // 공격을 시작하는 거리
    const float cAttackCooltime = 1f;
 
    NavMeshAgent _pathFinder;
    Transform _target;
    Stat _stat;
    private void Awake()
    {
        _pathFinder = GetComponent<NavMeshAgent>();
        _target = GameObject.FindGameObjectWithTag("Player").transform;
        _pathFinder.speed = Speed;
        _pathFinder.stoppingDistance = cDistance;
        _stat = Stat.Moving;
    }
    public override void Start()
    {
        base.Start();
        StartCoroutine(RefreshPath());
 
    }
 
    // Update is called once per frame
    public override void Update()
    {
        base.Update();
    }
 
    IEnumerator RefreshPath()
    {
        while(_target && !dead && _stat!=Stat.Attacking)
        {
            _pathFinder.SetDestination(_target.position);
            yield return new WaitForSeconds(0.5f);
        }
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
 
/// <summary>
/// 일정 방향으로 날아가며, 플레이어에게 부딪히면 데미지를 주고 사라지는 발사체
/// </summary>
 
public class Projectile : MonoBehaviour
{
    public int Atk;
    public int Speed;
    public float Distance;
 
    float _currentDistance; //현재 날아온 거리
 
    Player _target;
    
 
    private void FixedUpdate()
    {
        //최대 거리까지 날아왔으면 사라짐
        if (_currentDistance > Distance)
        {
            Destroy(gameObject);
        }
        Ray ray = new Ray(transform.position, transform.forward);
        RaycastHit hit;
 
        float movement = Speed * Time.deltaTime;//움직일 거리
        if (Physics.Raycast(ray, out hit, movement))
        {
            Debug.DrawRay(ray.origin, ray.direction);
            _target = hit.collider.GetComponent<Player>();
            if(_target != null)
            {
                CombatArgs args;
                args.AttackerAtk = Atk;
                args.DefenderDef = _target.Def;
                float damage = CombatSystem.CalculateInflictedDamage(args);
                _target.TakeHit((int)damage);
                Destroy(gameObject);
            }
            else
            {
                //계속 날아감.
                transform.Translate(Vector3.forward * movement);
                _currentDistance += movement;
            }
        }
        else
        {
            //계속 날아감.
            transform.Translate(Vector3.forward * movement);
            _currentDistance += movement;
 
        }
