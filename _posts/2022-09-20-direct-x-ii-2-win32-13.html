### 1. 애니메이터 클래스
\- 애니메이터는 GObject가 컴포넌트로 가지고 있으며, 애니메이터.Play(애니메이션 이름) 으로 원하는 애니메이션을 플레이할 수 있다.
\- Play 함수는 내부적으로 현재 애니메이션을 변경한다.
\- Update와 Render 함수를 갖는다. 현재 애니메이션의 Update와 Render 함수를 호출한다.
\- 애니메이션 생성 함수를 갖는다.
\- Update 구문에서 애니메이션이 반복재생인데 종료된 상태라면 리셋시켜서 다시 실행되게 한다.


```c++
//GObject.cpp
void GObject::FinalUpdate() {
	// 컴포넌트들에 대한 FinalUpdate들 모음.
	if (collider_ != nullptr) collider_->FinalUpdate();
	if (animator_ != nullptr) animator_->Update();
}

void GObject::ComponentRender(HDC hdc) {
	// 컴포넌트들에 대한 렌더링
	if (collider_ != nullptr) collider_->Render(hdc);
	if (animator_ != nullptr) animator_->Render(hdc);
}
```

```c++
//Animator.cpp
#include "pch.h"
#include "Animator.h"
#include "Animation.h"

Animator::Animator()
	: owner_(nullptr)
	, anims_{}
	, current_anim_(nullptr) {
}

Animator::Animator(const Animator& origin)
	: owner_(nullptr)
	, anims_(origin.anims_)
	, current_anim_(nullptr){
}

Animator::~Animator()
{
	//애니메이션들 삭제
	SafeDeleteMap<tstring, Animation*>(anims_);
}

void Animator::CreateAnimation(
	const tstring& name, Texture* texture, Vector2 base_pos, Vector2 img_size, Vector2 img_distance, Vector2 offset
	, float duration, int img_count, bool is_repeat)
{
	Animation* anim = new Animation();
	anim->Init(name, texture, base_pos, img_size, img_distance, offset, duration, img_count, is_repeat);
	anim->set_animator(this);
	anims_.insert(std::make_pair(name, anim));
}

void Animator::Play(const tstring& name)
{
	Animation* anim = FindAnimation(name);
	current_anim_ = anim;
}

Animation* Animator::FindAnimation(const tstring& name)
{
	std::map<tstring, Animation*>::iterator iter = anims_.find(name);
	
	if (iter == anims_.end()) return nullptr;
	return iter->second;
}

void Animator::Update()
{
	if (current_anim_ == nullptr) return;
	if (current_anim_->is_repeat() && current_anim_->is_finished()) {
		current_anim_->ResetFrame();
	}
	current_anim_->Update();
}

void Animator::Render(HDC hdc)
{
	if (current_anim_ == nullptr) return;
	current_anim_->Render(hdc);
}
```

### 2. 애니메이션 클래스와 애니메이션 프레임 구조체
\- 애니메이션: 달리기, 걷기 등의 한가지 연속된 동작을 담는 클래스.
\- 애니메이션 프레임: 연속된 동작 중의 딱 한 프레임을 담는 구조체.
\- 내부적으로 텍스쳐를 한 장 가지고 있다.
\- 초기화에 필요한 정보: 텍스쳐를 자르기 위한 시작 위치, 다음 시작 위치까지의 거리, 한 이미지의 크기, 자를 이미지 개수, 오브젝트와의 거리 오프셋, 한 프레임을 몇초동안 지속할지, 애니메이션을 반복재생할지 여부
\- 초기화 시 받은 정보로 애니메이션 프레임들을 만들어 가진다.
\- 0번 프레임부터 마지막 프레임까지 재생되면 애니메이션은 종료 상태가 된다.
\- 하나의 프레임은 시작 위치, 크기 정보를 갖는다.
\- Update 함수에서는 dt를 누적시켜 시간을 세면서 한 프레임의 유지시간이 지날 때마다 프레임 인덱스를 증가시키고, Render 함수에서는 현재 프레임을 그린다.
\- Render 함수에서 본인이 그려질 곳의 위치 정보를 알아야 하므로 animator를 알고 있어야 한다. animator는 owner를 알고 있으므로 animator->owner->get_pos로 오브젝트의 위치를 가져올 수 있게 된다.
\- 프레임 인덱스는 자를 이미지 개수보다 항상 작아야 한다. 프레임 인덱스가 자를 이미지 개수와 같거나 커지면 애니메이션은 종료 상태로 넘어간다. 종료상태일 땐 0번 프레임만 계속 보여준다.



```c++
//Animaion.h
#pragma once
class Texture;
class Animator;
struct AnimationFrame {
	Vector2 base_pos;
	Vector2 img_size;
};
class Animation
{
public:
	Animation();
	~Animation();
private:
	tstring name_;
	Texture* texture_;
	float duration_;
	bool is_repeat_;		//반복재생 여부
	Animator* animator_;
	int img_count_;
	Vector2 offset_;		//GObject와 실제 그려지는 애니메이션간의 거리
	bool is_finished_;		//재생이 끝났는지 여부

	float frame_acc_dt_;	//현재 프레임 실행 후 경과된 시간
	int frame_index_;

	std::vector<AnimationFrame> frames_;

public:
	void Init(tstring name, Texture* texture, Vector2 base_pos, Vector2 img_size, Vector2 img_distance, Vector2 offset
		, float duration, int img_count, bool is_repeat);
	void Update();
	void Render(HDC hdc);
	void ResetFrame();	//프레임을 0번째로 되돌리는 함수(재시작)
	inline void set_animator(Animator* animator) { animator_ = animator; };
	inline tstring& get_name() { return name_; };
	inline bool is_repeat() { return is_repeat_; };
	inline bool is_finished() { return is_finished_; };
	
};


```
```c++
//Animation.cpp
#include "pch.h"
#include "Animation.h"
#include "Animator.h"
#include "GObject.h"
#include "Texture.h"

Animation::Animation()
	: name_(_T(""))
	, texture_(nullptr)
	, offset_(Vector2{ 0, 0 })
	, duration_(0.)
	, is_repeat_(false)
	, animator_(nullptr)
	, frame_index_(0)
	, frame_acc_dt_(0.)
	, img_count_(0)
	, is_finished_(false)
{
}

Animation::~Animation()
{
}

void Animation::Init(tstring name, Texture* texture, Vector2 base_pos, Vector2 img_size, Vector2 img_distance
	, Vector2 offset, float duration, int img_count, bool is_repeat)
{
	name_ = name;
	texture_ = texture;
	duration_ = duration;
	is_repeat_ = is_repeat;
	img_count_ = img_count;
	offset_ = offset;

	//프레임 생성
	for (int i = 0; i < img_count; i++) {
		AnimationFrame frame{};
		frame.base_pos = base_pos + (img_distance)*i;
		frame.img_size = img_size;
		frames_.push_back(frame);
	}
}

void Animation::Update()
{
	if (is_finished_) return;

	//duration만큼 흘렀으면 다음 프레임으로 넘김
	frame_acc_dt_ += DtF();
	if (frame_acc_dt_ >= duration_) {
		if (++frame_index_ >= img_count_) {
			is_finished_ = true;
			frame_index_ = 0;
			return;
		}
		frame_acc_dt_ -= duration_;
	}
}

void Animation::Render(HDC hdc)
{

	//현재 프레임 그리기
	const Vector2& pos = animator_->get_owner()->get_pos() + offset_;
	const AnimationFrame& frame = frames_[frame_index_];
	TransparentBlt(
		hdc														//목적지 dc
		, static_cast<int>(pos.x - frame.img_size.x / 2.)		//left 좌표
		, static_cast<int>(pos.y - frame.img_size.y / 2.)		//top 좌표
		, static_cast<int>(frame.img_size.x)					//가로 길이
		, static_cast<int>(frame.img_size.y)					//세로 좌표
		, texture_->get_hdc()					//소스 dc(=비트맵이 선택된 메모리dc)
		, static_cast<int>(frame.base_pos.x)	//비트맵의 left 좌표
		, static_cast<int>(frame.base_pos.y)	//비트맵의 top 좌표
		, static_cast<int>(frame.img_size.x)	//비트맵의 가로 길이
		, static_cast<int>(frame.img_size.y)	//비트맵의 세로 길이
		, RGB(255, 0, 255));					//무시할 픽셀 색상

}

void Animation::ResetFrame()
{
	frame_index_ = 0;
	is_finished_ = false;
}

```




### 3. Player에 적용
\- 기존에 한 장의 텍스쳐를 그려줬던 것을, 애니메이터가 있다면 이 애니메이터의 현재 애니메이션을 재생하도록 변경한다.

```c++
//Player.cpp
	Texture* texture = ResManager::GetInstance()->LoadTexture(_T("player"), _T("texture\\MapleStory_Kino-Smaller.bmp"));
	Animator* animator = new Animator();
	animator->CreateAnimation(
		_T("Idle")
		, texture
		, Vector2{ 0, 0 }
		, Vector2{ 42, 42 }
		, Vector2{ 42, 0 }
		, Vector2{ 0, -15 }
		, 0.1f
		, 4
		, true);
	animator->set_owner(this);
	set_animator(animator);
    
	get_animator()->Play(_T("Idle"));
```
본인의 애니메이터를 지정하고, 애니메이터 안에 애니메이션을 추가한 뒤, 애니메이션을 실행해준다.

애니메이션에 사용할 텍스쳐는 포폴용 이미지를 제공하는 사이트에서 메이플스토리의 이미지를 가져왔다.
애니메이션의 오프셋을 {0, -15}로 잡아서 초기 위치를 Player의 pos보다 y축으로 -15만큼 위에 있도록 설정했다.
