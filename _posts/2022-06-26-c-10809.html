#include <iostream> int main(){ std::string S; std::cin>>S; int alphabetPos[26]; //각 알파벳의 첫 위치를 저장할 배열 std::fill_n(alphabetPos, 26, -1); //배열을 -1로 초기화 for(int i=0; i<S.length(); i++){ //각 단어를 첫 인덱스부터 돌면서 알파벳 배열에 위치 저장. if(alphabetPos[S[i]-'a'] == -1) alphabetPos[S[i]-'a'] = i; } for(int i=0; i<26; i++){ std::cout<<alphabetPos[i]<<" "; } }
