1. 접근하기

DFS와 BFS를 구현하면 되는 문제다. 연습차원에서 풀어봤다.

DFS는 스택으로, BFS는 큐로 구현했다.
DFS는 재귀함수로 구현하는 게 더 직관적이지만, 재귀함수보다 스택을 사용했을 때의 오버헤드가 적고 추가적인 작업을 하기 더 간편하므로(재귀함수는 전역변수를 사용해야 함) 스택을 사용했다.

#include <iostream>
#include <stack>
#include <queue>
bool g_map[1001][1001];
bool g_visited_dfs[1001];
bool g_visited_bfs[1001];
void Bfs(int node, int N) {
    std::queue<int> q;
    q.push(node);
    g_visited_bfs[node] = true;
    std::cout << node << " ";
    while (!q.empty()) {
        int cnode = q.front();
        q.pop();
 
        //cnode와 연결된 모든 노드를 큐에 넣기
        for (int i = 1; i <= N; i++) {
            if (!g_visited_bfs[i]
                && g_map[cnode][i]) {
                q.push(i);
                g_visited_bfs[i] = true;
                std::cout << i << " ";
            }
        }
    }
    std::cout << "\n";
 
}
void Dfs(int node, int N) {
    std::stack<int> stack;
 
    stack.push(node);
    g_visited_dfs[node] = true;
    std::cout << node << " ";
 
    //모든 연결된 노드를 탐색
    while (!stack.empty()) {
        int cnode = stack.top();
 
        bool all_visited = true;
        for (int i = 1; i <= N; i++) {
            if (!g_visited_dfs[i]
                && g_map[cnode][i]) {
                stack.push(i);
                g_visited_dfs[i] = true;
                std::cout << i << " ";
 
                //탐색하지 않은 항목이 있었으므로 false
                all_visited = false;
                break;
            }
        }
        //탐색하지 않은 자식이 없으면 탐색 완료로 치고 pop
        if (!all_visited) {
            stack.pop();
        }
 
    }
    std::cout << "\n";
 
}
int main() {
    //N:정점의 개수 M:간선의 개수 V:탐색 시작할 정점
    //map: N*N크기
    int N, M, V;
    std::cin >> N >> M >> V;
 
    for (int i = 0; i < M; i++) {
        int first, second;
        std::cin >> first >> second;
        g_map[first][second] = true;
        g_map[second][first] = true;
    }
 
    Dfs(V, N);
    Bfs(V, N);
 
 
}

풀면서 느낀 점이, BFS는 어차피 자식 항목을 한 번에 탐색하기 때문에 자식항목 탐색을 시작할 때 큐에서 부모 항목을 제거하지만,
DFS는 자식 항목에게 또 자식항목이 있을 경우 아래로 먼저 내려가서 탐색을 하기 때문에, 현재 항목의 최종 자식 항목까지 탐색했는지를 반드시 체크한 후 스택에서 빼야 한다.
